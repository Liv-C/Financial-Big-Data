myf = function(var,mpg)
{
  by = unique(mpg)
  mout = NULL
  for(i in by){
    s = subset(var, mpg == i)
    m = colMeans(s)
    mout = cbind(mout, m)
  }
  colnames(mout) = by
  mout
}
varlist = c("hwy","cty")
par(mar = c(4,4,6,3))
colspec = c(4,8)
tbl = myf(subset(mpg, , varlist), mpg$class)
